/* ================= SAFE URL BUILDER ================= */

function buildUrl(endpoint){
    if(endpoint.includes("?")){
        return `${BASE_URL}${endpoint}&api_key=${API_KEY}`;
    } else {
        return `${BASE_URL}${endpoint}?api_key=${API_KEY}`;
    }
}

/* ================= HERO SECTION - FIXED VERSION ================= */

function renderHero() {
    const main = document.getElementById("mainContent");
    
    const hero = document.createElement("div");
    hero.className = "hero";
    hero.id = "heroContainer";
    
    // Create all slides
    currentHeroMovies.forEach((movie, index) => {
        const slide = document.createElement("div");
        slide.className = "slide";
        if(index === 0) slide.classList.add("active");
        slide.style.backgroundImage = `url(${IMG + movie.backdrop_path})`;
        
        // Create overlay
        const overlay = document.createElement("div");
        overlay.className = "hero-overlay";
        
        const title = document.createElement("h1");
        title.innerText = movie.title;
        
        const desc = document.createElement("p");
        desc.innerText = movie.overview.substring(0, 160) + "...";
        
        const btn = document.createElement("button");
        btn.className = "btn";
        btn.innerText = "▶ Play";
        
        // DIRECT CLICK HANDLER - Store the movie ID directly
        const currentMovieId = movie.id;
        
        // Add click handler to the button
        btn.onclick = (function(id) {
            return function(e) {
                e.stopPropagation();
                console.log("Button clicked for movie ID:", id);
                openDetails(id);
            };
        })(currentMovieId);
        
        // Add click handler to the overlay (so clicking anywhere on the overlay works)
        overlay.onclick = (function(id) {
            return function(e) {
                // Don't trigger if clicking the button (already handled)
                if(e.target !== btn) {
                    console.log("Overlay clicked for movie ID:", id);
                    openDetails(id);
                }
            };
        })(currentMovieId);
        
        overlay.appendChild(title);
        overlay.appendChild(desc);
        overlay.appendChild(btn);
        slide.appendChild(overlay);
        hero.appendChild(slide);
    });
    
    main.appendChild(hero);
    
    // Start auto-slide
    if(heroInterval) clearInterval(heroInterval);
    
    let currentSlide = 0;
    const slides = hero.querySelectorAll(".slide");
    
    heroInterval = setInterval(() => {
        slides[currentSlide].classList.remove("active");
        currentSlide = (currentSlide + 1) % slides.length;
        slides[currentSlide].classList.add("active");
    }, 5000);
}

/* ================= RENDER ROW ================= */

async function renderRow(title, endpoint){

    const main = document.getElementById("mainContent");

    const data = await fetchData(buildUrl(endpoint));

    if(!data.results || data.results.length === 0) return;

    const section = document.createElement("section");

    section.innerHTML = `
        <h2 class="row-title">${title}</h2>
        <button class="row-arrow left">❮</button>
        <button class="row-arrow right">❯</button>
    `;

    const row = document.createElement("div");
    row.className = "movie-row";

    data.results.slice(0,25).forEach(movie => {

        if(!movie.poster_path) return;

        const card = document.createElement("div");
        card.className = "movie-card";

        const img = document.createElement("img");
        img.src = IMG + movie.poster_path;

        // Store movie ID in a closure
        const movieId = movie.id;
        img.addEventListener("click", (e) => {
            e.stopPropagation();
            openDetails(movieId);
        });

        const overlay = document.createElement("div");
        overlay.className = "overlay";

        const playBtn = document.createElement("button");
        playBtn.className = "btn";
        playBtn.innerText = "▶ Play";
        playBtn.addEventListener("click", (e) => {
            e.stopPropagation();
            openDetails(movieId);
        });

        const listBtn = document.createElement("button");
        listBtn.className = "btn";
        listBtn.innerText = "＋ My List";
        listBtn.addEventListener("click", (e) => {
            e.stopPropagation();
            addToList(movieId);
        });

        const rating = document.createElement("p");
        rating.innerText = `⭐ ${movie.vote_average}`;

        overlay.appendChild(playBtn);
        overlay.appendChild(listBtn);
        overlay.appendChild(rating);

        card.appendChild(img);
        card.appendChild(overlay);
        row.appendChild(card);

    });

    section.appendChild(row);
    main.appendChild(section);

    const left = section.querySelector(".left");
    const right = section.querySelector(".right");

    left.addEventListener("click",() => {
        row.scrollBy({left:-700, behavior:"smooth"});
    });

    right.addEventListener("click",() => {
        row.scrollBy({left:700, behavior:"smooth"});
    });
}

/* ================= TRAILER SYSTEM ================= */

function playTrailer(key){

    if(!key){
        alert("Trailer not available.");
        return;
    }

    const modal = document.getElementById("trailerModal");
    const player = document.getElementById("trailerPlayer");

    modal.style.display = "flex";

    player.innerHTML = `
        <iframe width="100%" height="100%"
        src="https://www.youtube.com/embed/${key}?autoplay=1"
        frameborder="0"
        allow="autoplay; encrypted-media"
        allowfullscreen>
        </iframe>
    `;
}

function closeTrailer(){
    const modal = document.getElementById("trailerModal");
    const player = document.getElementById("trailerPlayer");

    modal.style.display = "none";
    player.innerHTML = "";
}

/* ================= MY LIST SYSTEM ================= */

function addToList(id){

    let list = JSON.parse(localStorage.getItem("cinevibeList")) || [];

    if(!list.includes(id)){
        list.push(id);
        localStorage.setItem("cinevibeList", JSON.stringify(list));
        alert("Added to My List");
    } else {
        alert("Already in My List");
    }
}

async function showMyList(){

    const main = document.getElementById("mainContent");
    main.innerHTML = "<div style='padding:120px;'>Loading My List...</div>";

    let list = JSON.parse(localStorage.getItem("cinevibeList")) || [];

    if(list.length === 0){
        main.innerHTML = "<div style='padding:120px;'>Your list is empty. Add some movies from the home page!</div>";
        return;
    }

    main.innerHTML = `
        <section style="padding:100px 60px;">
            <h2>My List (${list.length} movies)</h2>
            <div style="display:flex; flex-wrap:wrap; gap:20px; margin-top:30px;" id="listContainer"></div>
        </section>
    `;

    const container = document.getElementById("listContainer");

    for(let id of list){

        const movie = await fetchData(buildUrl(`/movie/${id}`));

        const card = document.createElement("div");
        card.style.width = "200px";

        card.innerHTML = `
            <img src="${IMG + movie.poster_path}"
                 style="width:100%; height:300px; object-fit:cover; border-radius:14px; cursor:pointer;"
                 onclick="openDetails(${movie.id})">

            <button class="btn" style="margin-top:10px; width:100%;" onclick="removeFromList(${movie.id})">
                Remove from List
            </button>
        `;

        container.appendChild(card);
    }
}

function removeFromList(id){

    let list = JSON.parse(localStorage.getItem("cinevibeList")) || [];
    list = list.filter(movieId => movieId !== id);
    localStorage.setItem("cinevibeList", JSON.stringify(list));
    showMyList();
}

    const main = document.getElementById("mainContent");
    main.innerHTML = "<div style='padding:120px; text-align:center;'>Searching...</div>";

    const results = await fetchData(
        buildUrl(`/search/movie?query=${encodeURIComponent(query)}`)
    );

    if(results.results.length === 0) {
        main.innerHTML = "<div style='padding:120px; text-align:center;'>No movies found. Try another search!<br><br><button class='btn' onclick='loadHome()'>Back to Home</button></div>";
        return;
    }

    main.innerHTML = `
        <section style="padding:100px 60px;">
            <h2>Search Results for "${query}" (${results.results.length} found)</h2>
            <div style="display:flex; flex-wrap:wrap; gap:20px; margin-top:30px;">
                ${results.results.slice(0,40).map(movie=>`
                    <div style="width:200px; cursor:pointer;" onclick="openDetails(${movie.id})">
                        <img src="${movie.poster_path ? IMG + movie.poster_path : 'https://via.placeholder.com/200x300?text=No+Image'}"
                             style="width:100%; height:300px; object-fit:cover; border-radius:14px;">
                        <p style="margin-top:10px;">${movie.title}</p>
                    </div>
                `).join("")}
            </div>
        </section>
    `;
});
